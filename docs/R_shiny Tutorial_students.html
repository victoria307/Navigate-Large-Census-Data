1 🧪 R Shiny Tutorial

The Purpose of this slide is to help you with understanding the functions of R shiny and show you some examples about how to using it.

1.1 Set up

library(shiny)

1.2 Components

The components of R shiny can be divided into 3 parts:

  1. UI (User Interface) :

This is the part the user sees and interacts with. It includes:

  1. Inputs : Elements like sliders, text boxes, and dropdowns that collect user input.
  2. Outputs: Elements like plots, tables, and text that display results.

  1. Server : This is the backend logic that connects inputs and outputs.

  2. Reactivity : This is the system that enables automatic updates. When a user changes an input, the output updates automatically.

  3. Observer : Observers perform side effects in response to input changes but do not return values. For example, they can be used to display a notification or log a message when a user enters information.

ui <- fluidPage(
  # your layout and UI elements here
)

server <- function(input, output, session) {
  # your reactive logic here
}

shinyApp(ui = ui, server = server)

        sliderInput("slider2", label = h3("Slider Range"), min = 0,

  )

va_counties <- counties(state = "VA", year = 2022)

The education_data One of the column is the population of people with a high school diploma, and another column is the population who have a middle school diploma

# Set the library and install the packages

# Acquire the API by registering through https://api.census.gov/data/key_signup.html

census_api_key("Your own API", install = TRUE, overwrite = TRUE)
# Acquire the API by registering through https://api.census.gov/data/key_signup.html

#The main function to retrive the data here are https://cran.r-project.org/web/packages/tidycensus/tidycensus.pdf

#to check the name of the variable
v19 <- load_variables(2019, "acs5", cache = TRUE)
View(v19)

# for example, if I want to know the age
# here is the explanation of the codebook: https://data.census.gov/table/ACSDT5Y2022.B01001

va_2019 <- get_acs(geography = "county", 
              year = 2019,
              variables = c(total_population_25 = "B15003_001", high_school = "B15003_017", middle_school = "B15003_012"), 
              state = "VA",
              survey = "acs5",
              output = "wide")

education_data <- va_2019 %>%
  rename(total_population_25 = total_population_25E, 
         high_school = high_schoolE,
         middle_school = middle_schoolE) 

  #change the county name to make it match the one on the map
education_data$county_name <- gsub(", Virginia$", "", education_data$NAME)

Define the UI

ui <- fluidPage(
  titlePanel("Virginia Counties - Education Levels"),
  
  sidebarLayout(
    sidebarPanel(
      h4("Controls"),
      selectInput("education_type", 
                 "Show Education Level:",
                 choices = list("High School Diploma" = "high_school",
                               "Middle School Diploma" = "middle_school")),
      
      br(),
      h4("County Info"),
      textOutput("clicked_county"),
      textOutput("education_value")
    ),
    
    mainPanel(
      leafletOutput("map", height = "500px")
    )
  )
)

2.2.1 Define the Server

server <- function(input, output, session) {
  
  # Create the map
  output$map <- renderLeaflet({
    leaflet(va_counties) %>%
      addTiles() %>%
      setView(lng = -78.6569, lat = 37.4316, zoom = 7) %>%
      addPolygons(
        fillColor = "lightblue",
        weight = 2,
        opacity = 1,
        color = "white",
        fillOpacity = 0.7,
        layerId = ~NAME,  # This allows us to identify which county was clicked
        popup = ~paste("County:", NAME),
        highlight = highlightOptions(
          weight = 5,
          color = "#666",
          fillOpacity = 0.7,
          bringToFront = TRUE
        )
      )
  })
  
  observe({
    # Get the selected education data

    selected_values <- education_data[[input$education_type]]
    
    # Create a simple color function
    colors <- ifelse(va_counties$GEOID %in% education_data$GEOID, "red", "lightblue")
    
    # Update the map
    leafletProxy("map") %>%
      clearShapes() %>%
      addPolygons(
        data = va_counties,
        fillColor = colors,
        weight = 2,
        opacity = 1,
        color = "white",
        fillOpacity = 0.7,
        layerId = ~NAME,
        popup = ~paste("County:", NAME),
        highlight = highlightOptions(
          weight = 5,
          color = "#666",
          fillOpacity = 0.7,
          bringToFront = TRUE
        )
      )
  })
  
  # Show info when county is clicked
  output$clicked_county <- renderText({
    click <- input$map_shape_click
    if (is.null(click)) {
      "Click on a county"
    } else {
      paste("Selected:", click$id)
    }
  })
  
   output$education_value <- renderText({
    click <- input$map_shape_click
    if (is.null(click)) {
      "No data to show"
    } else {
      # Find the education data for this county
  # First get the GEOID of the clicked county
    clicked_county <- va_counties[va_counties$NAME == click$id, ]
    county_geoid <- clicked_county$GEOID

# Then match using GEOID
county_data <- education_data[education_data$GEOID == county_geoid, ]      
      if (nrow(county_data) > 0) {
        value <- county_data[[input$education_type]]
        paste(input$education_type, ":", value)
      } else {
        "No education data for this county"
      }
    }
  })
}

Run the application

shinyApp(ui = ui, server = server)

(i) **Inputs** : Elements like sliders, text boxes, and dropdowns that collect user input.
