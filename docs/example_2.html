years <- c(1980, 1990)

for(year in years) {
  # Load data
  data_i <- read_dta(paste0("data/outcome/", year, "_i.dta"))
  
  # Sort by serial and birthyr
  data_i %
    arrange(serial, birthyr)
  
  # Generate birth order for children with mothers in household
  data_i %
    group_by(serial) %>%
    mutate(birth_order = ifelse(momloc != 0, row_number(), NA)) %>%
    ungroup()
  
  # Create indicator for child's sex (1 for male, 0 for female)
  data_i %
    mutate(male = ifelse(sex == 1, 1, 0))
  
  # Generate first child sex indicator (earliest birth year)
  data_i %
    group_by(serial) %>%
    mutate(
      boy1st = case_when(
        birth_order == 1 & male == 1 ~ 1,
        birth_order == 1 & male == 0 ~ 0,
        TRUE ~ NA_real_
      )
    ) %>%
    ungroup()
  
  # Generate second child sex indicator
  data_i %
    group_by(serial) %>%
    arrange(birth_order) %>%
    mutate(
      boy2nd = case_when(
        birth_order == 2 & male == 1 ~ 1,
        birth_order == 2 & male == 0 ~ 0,
        TRUE ~ NA_real_
      )
    ) %>%
    ungroup()
  
  # Save data
  write_dta(data_i, paste0("data/outcome/", year, "_i.dta"))
}

3 Age at the first birth

for(year in years) {
  # Load data
  data_i <- read_dta(paste0("data/outcome/", year, "_i.dta"))
  
  # Calculate age at first birth
  data_i %
    arrange(serial, birth_order) %>%
    group_by(serial) %>%
    mutate(
      # Calculate age at first birth for first child
      agefstm = case_when(
        birth_order == 1 & !is.na(age_mom) & !is.na(age) ~ age_mom - age,
        TRUE ~ NA_real_
      )
    ) %>%
    # Fill down the age at first birth for other children in same household
    fill(agefstm, .direction = "down") %>%
    ungroup()
  
  # Save data
  write_dta(data_i, paste0("data/outcome/", year, "_i.dta"))
}

8.2 Handling API Rate Limits

# Function to geocode with automatic retries and rate limiting
geocode_with_retry <- function(address, max_retries = 3, delay = 1) {
  for(attempt in 1:max_retries) {
    tryCatch({
      result <- geocode(address, output = "more")
      if(!is.na(result$lat)) {
        return(result)
      }
    }, error = function(e) {
      if(attempt < max_retries) {
        cat("Attempt", attempt, "failed. Retrying in", delay, "seconds...\n")
        Sys.sleep(delay)
        delay <- delay * 2  # Exponential backoff
      }
    })
  }
  
  # Return NA if all attempts failed
  return(data.frame(lon = NA, lat = NA, address = NA))
}

11 Best Practices and Tips

11.1 1. API Key Security

  • Never hardcode API keys in scripts
  • Use environment variables: Sys.getenv("GOOGLE_MAPS_API_KEY")
  • Store keys in .Renviron file

11.2 2. Rate Limiting

  • Google allows 50 requests per second for geocoding
  • Add delays between requests for large batches
  • Consider using Sys.sleep() between geocoding calls

11.3 3. Address Formatting

  • Be consistent with address formats
  • Include city, state, and country when possible
  • Clean addresses before geocoding (remove special characters)

11.4 4. Error Handling

  • Always include try-catch blocks for large batches
  • Save intermediate results to avoid losing progress
  • Log failed geocoding attempts for manual review

11.5 5. Data Validation

  • Check for missing coordinates
  • Validate coordinates are within expected geographic bounds
  • Review geocoded addresses against original inputs
